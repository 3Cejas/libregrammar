--  NOUNS AND ADJECTIVES  --

    Noun:                   N [gender] [number] | NN
    Demonstrative           D [gender] [number] | DEM
    

                [gender]    M   = masculine
                            F   = feminine
                            E   = epicene

                [number]    S   = singular
                            T   = plural (two)
                            P   = plural
                            SP  = singular or plural

    Examples:
        NMS
        NFP

	
	Adjective:              ADJ


--  GRAMMATICAL WORDS  --

    preposition:                    PR


--  VERBS  --

    Verb:		             V[tense]
    
    			[tense]		P = past
    			
	Examples : 
		VP
     
    
``    Left open double quote
ØŒ     Comma
''    Right close double quote
.     Sentence-final punctuation (in LanguageTool, use SENT_END instead)
:     Colon, semi-colon
